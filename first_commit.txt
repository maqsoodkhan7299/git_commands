# to check the global config variable values like username emails etc
git config --list

# to change the global config variable values

# To check the git version
 git -v

# to initialize the git
git init

# to check the git status
git status

# add file to stagging area
git add first_commit.txt  "for single file to add in the staging
git add --all " for all files to add in the staging

#Note: Short status flags are:
git status --short or --long
"?? - Untracked files
"A - Files added to stage
"M - Modified files
"D - Deleted files

git log
